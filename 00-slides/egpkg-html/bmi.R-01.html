1 #' Body Mass Index
 2 #'
 3 #' Given weight and height, return the body mass index
 4 #'
 5 #' @param height height in inches
 6 #' @param weight weight in lbs
 7 #'
 8 #' @return a numeric vector of bmi
 9 #'
10 #' @examples
11 #' # The bmi of a six foot one inch tall, three hundred pound individual is
12 #' bmi(height = 6 * 12 + 1, 300)
13 #'
14 #' @export
15 bmi <- function(height, weight, hunits, wunits) {
16   h_meters <-
17     switch(hunits,
18            "in" = height * 0.0254,
19            "cm" = height * 100,
20            "feet" = height / 12 * 0.0254,
21            "m" = height)
22 
23   w_kg <-
24     switch(wunits,
25            "lbs" = weight * 0.453592,
26            "kg"  = weight)
27 
28   w_kg / h_meters ** 2
29 
30 }
